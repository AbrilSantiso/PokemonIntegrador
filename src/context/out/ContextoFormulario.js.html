import React from "react";
import { ReactElement } from "react";

export const ContextoFormulario = React.createContext();
/**
  Provider global del formulario de inscripción del Pokemon
  
  @author Abril
  @param {ReactElement} children Elemento jsx que tendra como hijo el provider
  @returns {ReactElement} Elemento Provider del contexto del formulario
 */
const ProviderFormulario = ({ children }) => {
  const [formulario, setFormulario] = React.useState({
    nombre: "",
    apellido: "",
    email: "",
  });

  const handleInputBlur = (valorInput) => {
    const { campo, valor } = valorInput;

    setFormulario({
      ...formulario,
      [campo]: valor,
    });
  };

  return (
    <ContextoFormulario.Provider
      value={{
        formulario,
        handleInputBlur,
      }}
    >
      {children}
    </ContextoFormulario.Provider>
  );
};

export default ProviderFormulario;
